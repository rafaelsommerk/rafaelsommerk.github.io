Ao dar uma olhada no código, você verá que a função random depende de um gerador de números pseudoaleatórios subjacente, que geralmente é baseado no algoritmo Mersenne Twister. Este é um algoritmo bem estudado e amplamente utilizado para geração de números pseudoaleatórios. O código no repositório fornecerá mais detalhes sobre a implementação

class MersenneTwister(seed: Long) {
private val w = 32
private val n = 624
private val m = 397
private val r = 31
private val a = 0x9908B0DF.toInt()
private val u = 11
private val d = 0xFFFFFFFF.toInt()
private val s = 7
private val b = 0x9D2C5680.toInt()
private val t = 15
private val c = 0xEFC60000.toInt()
private val l = 18
private val f = 1812433253

private val MT = IntArray(n)
private var index = n + 1
private val lowerMask = (1 shl r) - 1
private val upperMask = (1 shl r)

init {
seed(seed)
}

private fun seed(seed: Long) {
index = n
MT[0] = (seed and 0xFFFFFFFF).toInt()
for (i in 1 until n) {
MT[i] = ((f.toLong() * (MT[i - 1] xor (MT[i - 1] ushr (w - 2))) + i) and 0xFFFFFFFF).toInt()
}
}

private fun extractNumber(): Int {
if (index >= n) {
twist()
}

var y = MT[index]
y = y xor ((y ushr u) and d)
y = y xor ((y shl s) and b)
y = y xor ((y shl t) and c)
y = y xor (y ushr l)

index++
return y and 0xFFFFFFFF.toInt()
}

private fun twist() {
for (i in 0 until n) {
val x = (MT[i] and upperMask) or (MT[(i + 1) % n] and lowerMask)
val xA = x ushr 1 xor if (x % 2 != 0) a else 0
MT[i] = MT[(i + m) % n] xor xA
}

index = 0
}

// Exemplo de uso


fun generateNumbers(count: Int): List {
val numbers = mutableListOf()
repeat(count) {
numbers.add(extractNumber())
}
return numbers
}
}

fun main() {
val mt = MersenneTwister(seed = 42)
val generatedNumbers = mt.generateNumbers(10)
println(generatedNumbers)
}
